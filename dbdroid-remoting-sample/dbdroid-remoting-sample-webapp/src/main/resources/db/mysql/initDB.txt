CREATE DATABASE IF NOT EXISTS samplewebapp;
GRANT ALL PRIVILEGES ON samplewebapp.* TO root@localhost IDENTIFIED BY 'root';

USE samplewebapp;

CREATE TABLE CONTACTS (
	ID INT(4) UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	FIRSTNAME VARCHAR(50),
	LASTNAME VARCHAR(50),
	EMAIL VARCHAR(30),
	TELEPHONE VARCHAR(30)
) engine=InnoDB;
